📖 Webhook 설정 가이드 (Google Sheets 무료 방법)
1️⃣ Google Sheets 만들기

drive.google.com → 새 Google Sheets 생성 → 이름: "지웰홈스 신청내역"

2️⃣ Apps Script 열기

상단 메뉴 → 확장 프로그램 → Apps Script

3️⃣ 아래 코드 붙여넣기 
function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Sheet1') 
    || SpreadsheetApp.getActiveSpreadsheet().insertSheet('Sheet1');
  // 헤더 (첫 행이 비어있을 때만)
  if(sheet.getLastRow()===0){
    sheet.appendRow(['접수일시','유형','이름','연락처','이메일','룸타입','기간/시간','희망일','문의사항','상태','언어','ID']);
  }
  sheet.appendRow([
    new Date(data.createdAt),
    data.type === 'tour' ? '투어' : '계약',
    data.name, data.phone, data.email || '',
    data.roomType || '', data.period || data.time || '',
    data.date || '', data.note || '',
    data.status || '신청확인', data.lang || 'ko', data.id
  ]);
  // 선택: 본인 이메일로도 자동 전송
  // MailApp.sendEmail('your@email.com', '[지웰홈스] 신규 신청', JSON.stringify(data, null, 2));
  return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}
4️⃣ 배포 → 웹 앱

우측 상단 배포 → 새 배포 → 유형: 웹 앱 → 액세스 권한: 모든 사용자 → 배포 → URL 복사

5️⃣ 위 Webhook URL 칸에 붙여넣기 → 저장

이제 신청이 자동으로 Google Sheets로 들어옵니다! 📩

💡 대안: Slack Webhook, Discord Webhook, Make.com(통합), Zapier, Notion API 등 어떤 서비스든 가능합니다. JSON POST를 받는 URL이면 모두 작동합니다.
